# 数据结构

## 队列

SplPriorityQueue
```
    $pq = new SplPriorityQueue();

    $pq->insert('a', 10);
    $pq->insert('b', 1);
    $pq->insert('c', 8);

    echo $pq->count() .PHP_EOL; //3
    echo $pq->current() . PHP_EOL; //a

    /**
     * 设置元素出队模式
     * SplPriorityQueue::EXTR_DATA 仅提取值
     * SplPriorityQueue::EXTR_PRIORITY 仅提取优先级
     * SplPriorityQueue::EXTR_BOTH 提取数组包含值和优先级
     */
    $pq->setExtractFlags(SplPriorityQueue::EXTR_DATA);

    while($pq->valid()) {
        print_r($pq->current());  //a  c  b
        $pq->next();
    }
```

SplQueue
```
    $queue = new SplQueue();
    $queue->enqueue('A');
    $queue->enqueue('B');
    $queue->enqueue('C');

    $queue->rewind();
    while ($queue->valid()) {
        echo $queue->current(),"\n";//A B C
        $queue->next();
    }

    while (!$queue->isEmpty()) {
        echo $queue->dequeue(),"\n"; //A B C
    }
```

## 栈

SplStack
```
    $stack = new SplStack();

    $stack->push(1);
    $stack->push(2);
    $stack->push(3);
    $stack->push(4);
    $stack->push(5);

    $stack->rewind();

    while ($stack->valid()) {
        echo $stack->current(); //5 4 3 2 1
        $stack->next();
    }
```

## 双链表

SplDoublyLinkedList
```
        $dlist = new SplDoublyLinkedList();
    
        $dlist->push('hiramariam');
        $dlist->push('maaz');
        $dlist->push('zafar');
    
        /* the list contains
            hiramariam
            maaz
            zafar
        */
    
        $dlist->unshift(1);
        $dlist->unshift(2);
        $dlist->unshift(3);
    
        /* the list now contains
            3
            2
            1
            hiramariam
            maaz
            zafar
        */
    
        $dlist->pop();
    
        /* the list now contains
            3
            2
            1
            hiramariam
            maaz
    
        */
    
        $dlist->shift();
    
        /* the list now contains
    
            2
            1
            hiramariam
            maaz
    
        */
    
        $dlist->add(3 , 2.24);
    
        for($dlist->rewind();$dlist->valid();$dlist->next()){
    
            echo $dlist->current()."<br/>";
        }
    
        /* the list now contains
            2
            1
            hiramariam
            2.24
            maaz
        */
```
